# Como siempre, antes actualia zar el sistema
sudo apt-get update -y

# Preparar el entorno
cd ~/Downloads

# Crear un directorio temporal y descomprimir el fichero descargado:
mkdir temp
cd tempo
unzip bWAPPv2.2.zip   // o el nombre del fichero descargado


*************************************************
# Preparar la Base de Datos MySQL

mysql -V  // Comprobar si MySQL está instalada
    sudo apt install mysql-server // Instalar MySQL si no lo estuviera 

sudo systemctl start mysql    // arranca mysql
sudo systemctl enable mysql    // habilita MySQL en el arranque:
sudo systemctl status mysql    // checkea el stado de MySQL 
Q para salir.

*************************************************
# Prepartar instalación de servior web Apache2

apache2 -v    // comprueba si Apache2 está instalado
    sudo apt install apache2    // Instala Apache2 si no lo estuviera

sudo systemctl start apache2    // arranca Apache2:
sudo systemctl enable apache2    // habilita Apache2 en el arrnaque
sudo systemctl status apache2    // checkea el estado de Apache2
Q para salir


*************************************************
Preparar bWAPP

cd temp    // Navegar al directorio donde hemos descomprimido bWAPP
chmod 777 bWEAPP/     //Da permisos al directorio bWAPP
cd bWAPP 
Give all permissions to these directories:

chmod 777 passwords/
chmod 777 images/
chmod 777 documents/

Configure MySQL
Open MySQL to add a user and give privileges:

sudo mysql

Create a user (replace ahmed and pass123 with your desired username and password):

CREATE USER 'ahmed'@'localhost' IDENTIFIED BY 'pass123';

Grant privileges to the user you created:

GRANT ALL PRIVILEGES ON bWAPP.* TO 'ahmed'@'localhost';

Apply the changes immediately:

FLUSH PRIVILEGES;

Exit MySQL:

exit

Edit bWAPP Database Credentials
Edit the settings.php file:

mousepad admin/settings.php

Change root to the same username you created in MySQL:

$db_username = "ahmed";

Add the same password you created in MySQL:

$db_password = "pass123";

    Save changes (Ctrl + S), and close the file.

Important Step
Go to this link and copy the content (Ctrl + C).
Edit install.php:

mousepad install.php

    Select all (Ctrl + A), paste the copied content (Ctrl + V), save changes (Ctrl + S), and close the file.

Move bWAPP Files to Local Server
Navigate to the parent directory:

cd ../

Move all bWAPP files to the local server:

sudo mv * /var/www/html/

Install bWAPP Database

    Open the following link in your browser and click to install the database: http://localhost/bWAPP/install.php

Verify Database Installation
Open MySQL to check if the database has been added:

sudo mysql

Show databases:

SHOW DATABASES;

Exit MySQL:

exit
